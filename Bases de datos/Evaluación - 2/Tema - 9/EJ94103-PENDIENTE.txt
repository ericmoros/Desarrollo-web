  GROUP BY dept_no);

-- 100)	Visualiza el número de departamento del departamento que tenga más empleados.
SELECT dept_no, COUNT(*)
FROM emple
GROUP BY dept_no
HAVING COUNT(*) =
  (SELECT MAX(COUNT(*))
    FROM emple
    GROUP BY dept_no);

-- 101)	Buscar los departamentos que tienen más de dos personas trabajando en la misma profesión.
SELECT dept_no
FROM emple
WHERE oficio IN
  (SELECT oficio
    FROM emple
    GROUP BY oficio
    HAVING COUNT(*) > 2)
GROUP BY dept_no;
